126    def player_movement(self) -> Vector2:
127        """
128        Player movement module.
129
130        Takes in user input through pygame key wrappers.
131
132        Returns:
133            Vector2: resulting movement vector.
134        """
135        # don't move during action:
136        if self._action_a_cooldown > 0.0:
137            return Vector2()
138        # take in inputs
139        dir: Vector2 = Vector2()
140        if self._controller.right_movement:
141            dir.x += 1
142            self._orientation = True
143            self._curr_group = "E"
144        if self._controller.left_movement:
145            dir.x -= 1
146            self._orientation = False
147            self._curr_group = "W"
148        if self._controller.up_movement:
149            dir.y -= 1
150            self._curr_group = "N"
151        if self._controller.down_movement:
152            dir.y += 1
153            self._curr_group = "S"
154
155        return dir

169    def animate(self, time: float) -> None:
170        """
171        Player movement animation.
172
173        Using player positional vectors and previous positional vectors,
174        The player knows what direction to face, and what animations to use.
175
176        Args:
177            time (float): Time elapsed.
178        """
179        diff_x: float = self._position.x - self._prev_position.x
180        diff_y: float = self._position.y - self._prev_position.y
181
182        if not self._curr_orient == self._orientation and self._curr_group == "E":
183            self._curr_orient = not self._curr_orient
184            self.image = pygame.transform.flip(self.image, True, False)
185
186        self._anim_timer -= abs(diff_x) + abs(diff_y)
187
188        if self._anim_timer < 0:
189            self._anim_timer = self._anim_timer_top
190
191            if self._curr_step == "0":
192                self._curr_step = "1"
193            else:
194                self._curr_step = "0"
195
196        if self._action_a_cooldown > 0.0:
197            self._curr_step = "2"
198        elif self._curr_step == "2":
199            self._curr_step = "1"
200
201        self.image = self._assets[f"{self._curr_group}move{self._curr_step}"]

295    def handle_inputs(self) -> None:
296        """
297        Take in all key and controller inputs (if there is one)
298
299        For movement, check all valid inputs.
300        """
301        # reset all values
302        self.reset_values()
303        # controller input gathering
304        self._handle_joystick_inputs()
305        # keyboard input gathering
306        self._handle_keyboard_inputs()
307
308    def _handle_joystick_inputs(self) -> None:
309        """
310        Take in controller inputs.
311        """
312        if not self._joystick:
313            return
314
315        try:
316            jaxis_0: float = self._joystick.get_axis(0)  # left -> right
317            jaxis_1: float = self._joystick.get_axis(1)  # up -> down
318            dpad_up: int = self._joystick.get_button(11)
319            dpad_down: int = self._joystick.get_button(12)
320            dpad_left: int = self._joystick.get_button(13)
321            dpad_right: int = self._joystick.get_button(14)
322            self._action_a = self._joystick.get_button(0)
323            self._action_b = self._joystick.get_button(1)
324        except pygame.error:
325            raise pygame.error("Failed to get controller.")
326
327        if jaxis_0 > self.AXIS_T or dpad_right:
328            self._right_movement = True
329        elif jaxis_0 < -self.AXIS_T or dpad_left:
330            self._left_movement = True
331        if jaxis_1 < -self.AXIS_T or dpad_up:
332            self._up_movement = True
333        elif jaxis_1 > self.AXIS_T or dpad_down:
334            self._down_movement = True
335
336    def _handle_keyboard_inputs(self) -> None:
337        """
338        Take in keyboard inputs.
339        """
340        keys: key.ScancodeWrapper = key.get_pressed()
341        if self.__check_key(keys, self.LEFT_KEYS):
342            self._left_movement = True
343        if self.__check_key(keys, self.RIGHT_KEYS):
344            self._right_movement = True
345        if self.__check_key(keys, self.UP_KEYS):
346            self._up_movement = True
347        if self.__check_key(keys, self.DOWN_KEYS):
348            self._down_movement = True
349
350        if self.__check_key(keys, self.ACTION_A_KEY):
351            self._action_a = True
352        if self.__check_key(keys, self.ACTION_B_KEY):
353            self._action_b = True
354
355    def reset_values(self) -> None:
356        self._up_movement = False
357        self._down_movement = False
358        self._left_movement = False
359        self._right_movement = False
360        self._up_movement = False
361        self._down_movement = False
362        self._action_a = False
363        self._action_b = False
364
365    def __check_key(self, keys: key.ScancodeWrapper,
366                    key_codes: list[int]) -> bool:
367        for code in key_codes:
368            if keys[code]:
369                return True
370        return False
371
372    @staticmethod
373    def controller_status() -> bool:
374        """
375        Check a controller's connection status.
376        """
377        return True if pygame.joystick.get_count() > 0 else False

316        try:

318            jaxis_1: float = self._joystick.get_axis(1)  # up -> down
319            dpad_up: int = self._joystick.get_button(11)
320            dpad_down: int = self._joystick.get_button(12)
321            dpad_left: int = self._joystick.get_button(13)
322            dpad_right: int = self._joystick.get_button(14)
323            self._action_a = self._joystick.get_button(0)
324            self._action_b = self._joystick.get_button(1)
325        except pygame.error:
326            raise pygame.error("Failed to get controller.")
327
328        if jaxis_0 > self.AXIS_T or dpad_right:
329            self._right_movement = True
330        elif jaxis_0 < -self.AXIS_T or dpad_left:
331            self._left_movement = True
332        if jaxis_1 < -self.AXIS_T or dpad_up:
333            self._up_movement = True
