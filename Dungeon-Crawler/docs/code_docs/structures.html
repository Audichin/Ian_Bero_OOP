112        This is a list representation of all the enemies
113        belonging to this room.
114        """

116            "Jelly", "Jelly", "Jelly", "Jelly", "Jelly",
117            "Coral", "Coral", "Coral",

197    def update_puzzle(self) -> None:
198        """
199        Updates the room if it is a puzzle room.
200        """
201        # check if enemies are alive
202        if len(self._enemies):
203            return
204
205        # check enemy patterns
206        if self._current_pattern >= len(self._puzzle_enemy_pattern):
207            return
208
209        # pop next pattern
210        next_pattern = self._puzzle_enemy_pattern[self._current_pattern]
211        self._current_pattern += 1
212        for enemy in next_pattern:
213            self.create_enemy(enemy['name'], enemy['position'])
214
215        return

329    def _generate_layout(self) -> None:
330        """
331        Generates the layout for dungeon
332        """
333        start = Room(self._world, 0, 0, self._rng, "start")
334        self.rooms[(0, 0)] = start
335        active_rooms = [start]
336
337        directions = [(0, 1), (0, -1), (1, 0), (-1, 0)]
338
339        while len(self.rooms) < self._total_rooms:
340            current = self._rng.choice(active_rooms)
341            dx, dy = self._rng.choice(directions)
342
343            new_x = current.x + dx
344            new_y = current.y + dy
345
346            if (new_x, new_y) not in self.rooms:
347                new_room = Room(self._world, new_x, new_y, self._rng)
348                self.rooms[(new_x, new_y)] = new_room
349                current.connect(new_room)
350                active_rooms.append(new_room)

352    def _find_farthest_room(self, start: Room) -> Room:
353        """
354        Finding the farthest room for setting the dungeon room to the farthest room away.
355
356        Args:
357            start (Room): The starting room (usually the one at (0, 0)).
358        Returns:
359            Room: The farthest room from the starting room.
360        """
361        visited: set[Room] = set()  # no duplicate items
362        queue: deque[tuple[Room, int]] = deque([(start, 0)])
363        farthest: tuple[Room, int] = (start, 0)
364
365        while queue:
366            room, dist = queue.popleft()
367            visited.add(room)
368
369            if dist > farthest[1]:
370                farthest = (room, dist)
371
372            for neighbor in room.connections:
373                if neighbor not in visited:
374                    queue.append((neighbor, dist + 1))
375
376        return farthest[0]

378    def _assign_room_types(self) -> None:
379        """
380        Assigns room types to the active rooms list, required to be called after _generate_layout()
381        """
382        all_rooms = list(self.rooms.values())
383
384        start_room = self.rooms[(0, 0)]
385
386        # Find farthest room for boss
387        boss_room = self._find_farthest_room(start_room)
388        boss_room.room_type = "boss"
389        boss_room.create_enemy('Boss', pygame.Vector2(
390            Room.SCREEN_CENTER[0] + 300, Room.SCREEN_CENTER[1]))
391
392        remaining = [
393            r for r in all_rooms
394            if r != start_room and r != boss_room
395        ]
396
397        # Assign puzzle rooms
398        puzzle_rooms = self._rng.sample(
399            remaining,
400            min(self._min_puzzle_rooms, len(remaining))
401        )
402
403        for room in puzzle_rooms:
404            room.room_type = "puzzle"
405            room.init_puzzle_patterns()
406
407        # Assign enemy rooms to remaining empty rooms
408        for room in remaining:
409            if room.room_type == "empty":
410                room.room_type = "enemy"
411                room.init_enemies()

413    def wall_hitbox(self, room: Room, orientation: str) -> list[pygame.Rect]:
414        """
415        Checks if the wall selected (depending on orientation)
416        has a door and is open, returning two types of hitboxes for
417        that wall for 3 different scenarios:
418        1) Door = true, state = closed -> fully blocked wall
419        2) Door = false -> fully blocked wall
420        3) Door = true, state = open -> two walls on either side with a
421                                        collider in the middle
422                                        to allow player into next room
423
424        args:
425            room (Room): the room we are checking
426            orientation (str): the wall orientation we are checking (W, N, E, S)
427        returns:
428            list[pygame.Rect]: All rects of the wall corresponding to orientation.
429        """
430        wall_data = self._generation.room_walls.get((room.x, room.y, orientation))
431        if wall_data is None:
432            # Refresh once in case generation data is stale.
433            self._generation.Apply_textures()
434            wall_data = self._generation.room_walls.get((room.x, room.y, orientation))
435        if wall_data is None:
436            raise ValueError(
437                f"No wall data for room ({room.x}, {room.y}) orientation {orientation}")
438
439        hasdoor = wall_data["hasdoor"]
440        isopen = wall_data["isopen"]
441
442        if orientation == "N":
443            return self.get_wall_N_rects(hasdoor, isopen)
444        elif orientation == "S":
445            return self.get_wall_S_rects(hasdoor, isopen)
446        elif orientation == "E":
447            return self.get_wall_E_rects(hasdoor, isopen)
448        elif orientation == "W":
449            return self.get_wall_W_rects(hasdoor, isopen)
450
451        raise ValueError(f"Invalid orientation for wall hitbox: {orientation}")

522        # wall attribute get
523        hasdoor = wall_data['hasdoor']
524        isopen = wall_data['isopen']

527        if not hasdoor:
528            return  # No changes

531        if walltype.__str__() == "Boss" and not boss:

534        wall_data['isopen'] = not isopen  # flip state
535
536        # set image path
537        image_path = f"{self._generation.PROJECT_DIR_NAME}/assets/visual/textures/walls"
538        if walltype.__str__() == "Boss":
539            image_path += "/boss/"
540        else:
541            image_path += "/door/"
542
543        # set the rest of the path corresponding to the wall.
544        image_path += orientation + "_"
545        if wall_data['isopen']:
546            image_path += "o"
547        else:
548            image_path += "x"
549
550        if orientation == "S" and walltype.__str__() == 'Boss':
551            image_path += "_" + walltype.__str__()
552        elif orientation != "S":
553            image_path += "_" + walltype.__str__()
