149    def loop(self, delta: float) -> None:
150        """
151        World Loop method.
152        > Once per frame, loop() should be called. In a game loop, all logic is processed.
153
154        Processed logic:
155        - Room changes / updates
156        - player loop
157        - entity loop
158        - UI changes / updates
159        - etc
160
161        Args:
162            delta (float): delta time. Milliseconds passed since last frame.
163        """
164
165        # Stop game if player health is zero
166        if self._player.HP <= 0:
167            if self._curr_room.room_type == "boss":
168                self._curr_room.enemies[0].HP = 15
169                self._curr_room = self._prev_room
170                self._player.position = pygame.Vector2(self.SCREEN_CENTER[0], self.SCREEN_CENTER[1])
171                self._player.HP = 8
172                self._sound_manager.play_audio(5)
173                return
174            self._player.position = pygame.Vector2(-999, -999)
175            return
176
177        if self._victory:
178            self._ui.update_victory(self._victory)
179            return
180
181        self._time += delta
182        self._player.loop(delta)
183
184        for indx, entity in enumerate(self._curr_room.enemies):
185            entity.loop(delta)
186            if entity.HP <= 0:
187                self._curr_room.enemies.pop(indx)
188
189        if len(self._curr_room.items):
190            for item in self._curr_room.items:
191                item.loop(delta)
192
193        self._item_slot.loop(delta)
194
195        self.update_room(delta)
196        self.update_ui()

200    def render(self) -> list[tuple[pygame.surface.Surface, pygame.rect.Rect]]:
201        """World Render method.
202        > Called once at the end of each frame by game, **AFTER** loop.
203
204        > When called, all render functions of every object should be called.
205
206        > Sounds should be played and music should be updated.
207
208        To render:
209        - sounds
210        - Room
211        - entities
212        - items
213        - UI
214        - etc.
215
216        Returns:
217            list[tuple[pygame.surface.Surface, pygame.rect.Rect]]: Render data to blit.
218        """
219        temp: list[tuple[pygame.Surface, pygame.Rect]] = []
220
221        for to_render in self.render_room():
222            temp.append(to_render)
223
224        temp.append(self._player.render(self._time)[0])
225        for entity in self._curr_room.enemies:
226            for entity_item in entity.render(self._time):
227                temp.append(entity_item)
228
229        if len(self._curr_room.items):
230            for item in self._curr_room.items:
231                temp.append(item.render())
232
233        for proj in self._item_slot.render_projectiles():
234            temp.append(proj)
235
236        # black box for transition
237        if self._transition_state:
238            # create a black box
239            black_box: pygame.Surface = pygame.Surface((1440, 810))
240            black_box.fill((0, 0, 0))
241            black_box.set_alpha(int(self._room_transition * 255))
242            temp.append((black_box, black_box.get_rect()))
243
244        # final one on display stack
245        for elem in self._ui.render():
246            temp.append(elem)
247
248        self.play_world_music()
249        self.play_sound_effects()
250        return temp

265    def play_world_music(self) -> None:
266        """
267        Sets the world music according to the room type.
268        > This method should change the music once the room type changes
269
270        > Example: Enemy -> Puzzle
271        """
272        # print(f"Music curr_room: {self._curr_room.room_type}, \
273        #       Music prev_room: {self._prev_room_type}")
274        if self._curr_room.room_type != self._prev_room_type:
275            prev_song = self._prev_music.pop()
276            # print(f"Music curr_song: {self._Music_IDs[self._curr_room.room_type]}")
277            self._sound_manager.stop_audio(prev_song)  # stop previous music
278            # play new music
279            self._sound_manager.play_audio(self._Music_IDs[self._curr_room.room_type])
280            # update prev_music
281            self._prev_music.append(self._Music_IDs[self._curr_room.room_type])
282            self._prev_room_type = self._curr_room.room_type
283        else:
284            pass

299    def update_room(self, delta: float) -> None:
300        """Updates the room according to changes.
301
302        > Called each frame, updates the room according to puzzle completion,
303
304        > door unlocks, etc.
305
306        Args:
307            delta (float): Delta time.
308        """
309        # puzzle room
310        match self._curr_room.room_type:
311            case "puzzle":
312                # get puzzle room state
313                puzzle_room_state = self._curr_room.puzzle_state
314
315                # Puzzle room beat!
316                if puzzle_room_state == 2:
317                    self._curr_room.room_type = "start"
318                    self._curr_room.create_item('keyfragment', pygame.Vector2(
319                        self.SCREEN_CENTER[0], self.SCREEN_CENTER[1]
320                    ))
321                    self._dungeon.set_all_doors_in_room(self._curr_room, True, False)
322
323                # transition when all enemies beat
324                elif puzzle_room_state == 1 and not self._transition_state:
325                    self._transition_state = 1
326
327                # set next enemies, place player at center
328                elif puzzle_room_state == 1 and self._room_transition >= 1:
329                    self._player.position = pygame.Vector2(
330                        self.SCREEN_CENTER[0], self.SCREEN_CENTER[1]
331                    )
332                    self._curr_room.create_item('heart', pygame.Vector2(
333                        self.SCREEN_CENTER[0], self.SCREEN_CENTER[1]
334                    ))
335                    self._curr_room.update_puzzle()
336
337            # place a heart in center of room when all enemies beat in
338            # an enemy room
339            case "enemy":
340                if not self._curr_room.enemies and not self._curr_room.room_clear:
341                    self._curr_room.create_item('heart', pygame.Vector2(
342                        self.SCREEN_CENTER[0], self.SCREEN_CENTER[1]
343                    ))
344                    self._curr_room.room_clear = True
345
346            case "boss":
347                # check that the boss is defeated
348                if not self._curr_room.enemies:
349                    self._victory = True
350                    self._curr_room.room_type = "start"
351
352        # check if the player has the key to open the boss door
353        if self._inventory and isinstance(self._inventory[0], Key):
354            if self._curr_room.room_type != "boss":
355                self._dungeon.set_boss_door_in_room(self._curr_room, True)
356
357        # handle transition
358        if self._transition_state:
359            self._room_transition += (delta * self._transition_state) * 10
360
361        if self._room_transition > 1 and self._transition_state == 1:
362            self._transition_state = -1
363        elif self._room_transition <= 0 and self._transition_state == -1:
364            self._transition_state = 0
365            self._room_transition = 0

414    def switch_room(self, cardinal: str) -> None:
415        """Go to the next room with respects to the cardinal of the door
416        the player interacted with.
417
418        Args:
419            cardinal (str): Cardinal direction to switch room towards.
420
421        Raises:
422            KeyError: Room attempted to be accessed doesn't exist.`
423        """
424        # set transition state to 1
425        if not self._transition_state:
426            self._transition_state = 1
427
428        if self._room_transition < 1:
429            return
430
431        position_tp: dict[str, tuple[float, float]] = {
432            'N': (720, 725),  # position to tp player if they go through N door
433            'E': (240, 445),  # position to tp player if they go through E door
434            'S': (720, 150),  # position to tp player if they go through S door
435            'W': (1200, 445)  # position to tp player if they go through W door
436        }
437
438        self._prev_room = self._curr_room
439        next_room: tuple[int, int] = (0, 0)
440        match cardinal:
441            case 'S':  # going through south door
442                next_room = (self._curr_room.x, self._curr_room.y - 1)
443            case'N':  # going through north door
444                next_room = (self._curr_room.x, self._curr_room.y + 1)
445            case 'E':  # going through east door
446                next_room = (self._curr_room.x + 1, self._curr_room.y)
447            case 'W':  # going through west door
448                next_room = (self._curr_room.x - 1, self._curr_room.y)
449
450        try:
451            self._curr_room = self._dungeon.rooms[next_room]
452        except KeyError:
453            raise KeyError("room doesnt exist")
454
455        match self._curr_room.room_type:
456            case"puzzle":
457                self._player.position = pygame.Vector2(
458                    self.SCREEN_CENTER[0], self.SCREEN_CENTER[1])
459                self._dungeon.set_all_doors_in_room(self._curr_room, False, False)
460                return
461            case "boss":
462                self._player.position = pygame.Vector2(
463                    self.SCREEN_CENTER[0] - 300, self.SCREEN_CENTER[1])
464                self._dungeon.set_all_doors_in_room(self._curr_room, False, True)
465                return
466        # teleport player to appropriate position
467        self._player.position = pygame.Vector2(
468            position_tp[cardinal][0],
469            position_tp[cardinal][1])

514    def entity_action(self, entity: Entity, action: str,
515                      projectile: Projectile | None = None) -> Any:
516        """Get entity actons and change the world accordingly.
517        > Whenever an entity makes an action (such as attacking)
518
519        > this function is to be called by that entity.
520
521        requests:
522        * s_col: Static collision
523        * player_pos: get player position
524        * player_col: check this for collision with player
525        * player_dmg_1: damage the player by one point
526        * player_dmg_2: damage the player by two points
527
528        Args:
529            entity (Entity): Entity calling to world.
530            action (str): Action entity wishes to call.
531            projectile (Projectile | None, optional): Projectiles connected. Defaults to None.
532
533        Returns:
534            Any: Status, arguments, etc.
535        """
536        # this collision return is temporary.
537        # s_col returns all static objects (walls, pits, etc)
538        match action:
539            case "s_col":
540                if projectile:
541                    return self._static_collision(projectile)
542                return self._static_collision(entity)
543            case "player_pos":
544                return self._player.position
545            case "player_col":
546                if projectile:
547                    if pygame.sprite.collide_rect(projectile, self._player):
548                        return self._player.rect
549                elif pygame.sprite.collide_rect(entity, self._player):
550                    return self._player.rect
551            case "player_dmg_1":
552                self._player.damage(1)
553            case"player_dmg_2":
554                self._player.damage(2)
555
556        return 0

558    def _static_collision(self, obj: Entity | Projectile) -> list[pygame.Rect]:
559        """Check entity collides with nearest wall
560
561        Args:
562            obj (Entity | Projectile): Object to pass static collision data to.
563
564        Returns:
565            list[pygame.Rect]: Collision data (passed if object collides.)
566        """
567        collides: list[pygame.Rect] = list[pygame.Rect]()
568        checks: list[str] = []
569
570        # Check position relative to Y direction
571        if obj.position.y < self.SCREEN_CENTER[1]:
572            checks.append('N')
573        elif obj.position.y > self.SCREEN_CENTER[1]:
574            checks.append('S')
575        # Check position relative to X direction
576        if obj.position.x > self.SCREEN_CENTER[0]:
577            checks.append('E')
578        elif obj.position.x < self.SCREEN_CENTER[0]:
579            checks.append('W')
580
581        dungeon_walls: dict[str, list[pygame.Rect]] = self.get_room_hitboxes()
582        for cardinal in checks:
583            for segment, rec in enumerate(dungeon_walls[cardinal]):
584                if obj.rect.colliderect(rec):
585                    if segment == 1 and obj is self._player:
586                        self.switch_room(cardinal)
587                        return []
588                    collides.append(rec)
589
590        return collides

594    def item_action(self, item: Item, action: str,
595                    projectile: Projectile | None = None) -> Any:
596        """
597        Get item action and projectiles.
598
599        requests:
600        * grabbed: called when item is touched
601        * store: called when item is put into inventory
602        * use: called when the item should be used and deleted
603        * heal_1: used to heal the player by one point
604        * attack: used by item slot items that act as weapons
605        * s_col: used for projectiles
606
607        Args:
608            item (Item): Item requesting action.
609            action (str): Action code.
610            projectile (Projectile | None, optional): Projectile passed. Defaults to None.
611        """
612        match action:
613            case "grabbed":
614                if pygame.sprite.collide_rect(item, self._player):
615                    return True
616            case "store":
617                if pygame.sprite.collide_rect(item, self._player):
618                    self._inventory.append(item)
619                    self._curr_room.items.remove(item)
620
621                    # check if we have four fragments
622                    if len(self._inventory) == 4:
623                        self._inventory.clear()
624                        self._inventory.append(Key(self))
625
626                    return None
627            case "use":
628                self._curr_room.items.remove(item)
629                return None
630            case "heal_1":
631                self._player.HP += 1
632                return None
633            case "attack":
634                for entity in self._curr_room.enemies:
635                    if projectile and pygame.sprite.collide_rect(projectile, entity):
636                        entity.damage(projectile.damage_points)
637                        return True
638            case "s_col":
639                if projectile:
640                    return self._static_collision(projectile)
641        return 0
